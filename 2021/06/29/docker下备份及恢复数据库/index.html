docker下自动备份及恢复数据库

#!/bin/bash
docker exec -i mysql8 bash <<'EOF'
echo "数据库开始备份"
mysqldump -u 用户名 -h 远程地址 -p"密码" --set-gtid-purged=off 库名 > ~/liang.sql
echo "数据库备份完成"
echo "准备同步数据库"
mysql -u 用户名 -p"密码" 恢复的数据库名称 < ~/liang.sql
echo "数据库同步完成"
rm -f ~/liang.sql
exit
EOF
